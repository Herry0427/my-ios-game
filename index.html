async function initSystem() { const statusLabel = document.getElementById('status-text'); try { // 1. 检查 SDK 是否加载成功 statusLabel.innerText = "自检：检查 SDK..."; if (!window.supabase) { alert("报错：库文件 supabase.js 加载失败！请检查仓库里是否有这个文件。"); return; } // 2. 初始化客户端 statusLabel.innerText = "自检：连接云端..."; // 确保这里的 URL 和 Key 是你新加坡新项目的！ supabase = window.supabase.createClient(SB_URL, SB_KEY); // 3. 读取本地数据 statusLabel.innerText = "自检：读取本地..."; const local = localStorage.getItem('my_game_center_profile'); if (local) userProfile = JSON.parse(local); updateUI(); // 4. 请求云端 statusLabel.innerText = "自检：正在握手..."; const { data, error } = await supabase .from('profiles') .select('*') .eq('id', userProfile.id) .single(); if (error) { if (error.code === 'PGRST116') { statusLabel.innerText = "同步：创建新档案..."; await supabase.from('profiles').insert([ { id: userProfile.id, nickname: userProfile.nickname, high_score: userProfile.highScore } ]); setStatus(true, "云端连接成功"); } else { // 如果这里报错，通常是 RLS 没关或者表名写错 alert("云端报错代码: " + error.code + "\n信息: " + error.message); setStatus(false, "连接报错"); } } else { statusLabel.innerText = "同步：同步数据..."; userProfile.highScore = Math.max(userProfile.highScore, data.high_score); localStorage.setItem('my_game_center_profile', JSON.stringify(userProfile)); setStatus(true, "云端同步完成"); } updateUI(); } catch (err) { alert("系统崩溃报错: " + err.message); } }
